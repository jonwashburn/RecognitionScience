\section{Basic Definitions}
\begin{definition}[Variety]
Smooth projective complex variety of dimension $n$.
\end{definition}

\begin{definition}[Hodge Class]
Element of $H^{p,p}(X,\mathbb{C}) \cap H^{2p}(X,\mathbb{Z})$.
\end{definition}

\begin{definition}[Rational Hodge Class]
Element of $H^{p,p}(X,\mathbb{Q}) = H^{2p}(X,\mathbb{Q}) \cap H^{p,p}(X,\mathbb{C})$.
\end{definition}

\section{Ledger Embedding}

\begin{definition}[Denominator Spectrum]
For $\alpha \in H^{p,p}(X,\mathbb{Q})$: $\text{DenSpec}(\alpha) = \{q_i : a_i \neq 0\}$.
\end{definition}

\begin{definition}[Hodge Hilbert Space]
$\mathcal{H} = \ell^2(\mathcal{P}, w_\varepsilon)$ where $w_\varepsilon(q) = q^\varepsilon$, $\varepsilon = \varphi - 1$.
\end{definition}

\begin{definition}[Evolution Operator]
$A_{n,\varepsilon}(s): \mathcal{H} \to \mathcal{H}$, $(A_{n,\varepsilon}(s) f)(q) = q^{-(s-(n+1)+\varepsilon)} f(q)$.
\end{definition}

\begin{lemma}[Operator Properties]
\begin{enumerate}
\item $\|A(s)\| = 2^{-((\text{Re}(s)-(n+1))+\varepsilon)}$
\item $A(s)$ is Hilbert-Schmidt iff $\text{Re}(s) > n + \frac{3-\sqrt{5}}{4}$
\item For real $s$, all eigenvalues are positive
\end{enumerate}
\end{lemma}

\begin{theorem}[Determinant-Zeta Identity]
$\det_2(I - A(s)) \cdot E_{n,\varepsilon}(s) = \zeta_{\text{Hdg}}(s)^{-1}$
\end{theorem}

\section{Positivity}

\begin{theorem}[Positivity Off Critical Plane]
For $\text{Re}(s) > n+1$, the operator $I - A(s)$ is positive definite.
\end{theorem}

\begin{lemma}[Spectral Gap]
$\text{gap}(I - A(s)) = 1 - 2^{-((\text{Re}(s)-(n+1))+\varepsilon)} > 0$
\end{lemma}

\begin{corollary}[Determinant Non-Zero]
For $\text{Re}(s) > n+1$: $\det_2(I - A(s)) > 0$.
\end{corollary}

\section{Functional Equation}

\begin{theorem}[Eight-Beat Periodicity]
$\Theta^8 = I$ where $\Theta$ is the phase map.
\end{theorem}

\begin{theorem}[Functional Equation]
$\zeta_{\text{Hdg}}(s) = \chi(s) \cdot \zeta_{\text{Hdg}}(2(n+1) - s)$ with $|\chi(s)| = 1$ on critical line.
\end{theorem}

\begin{corollary}[Zero Symmetry]
If $\zeta_{\text{Hdg}}(s_0) = 0$ then $\zeta_{\text{Hdg}}(2(n+1) - s_0) = 0$.
\end{corollary}

\section{Main Theorem}

\begin{theorem}[Zeros on Critical Line]
All zeros of $\zeta_{\text{Hdg}}(s)$ with $\text{Re}(s) > 0$ lie on $\text{Re}(s) = n+1$.
\end{theorem}

\begin{lemma}[Zero-Class Correspondence]
$\alpha$ is algebraic iff its contribution to $\zeta_{\text{Hdg}}$ has all zeros on critical line.
\end{lemma}

\begin{theorem}[Hodge Conjecture]
Every rational $(p,p)$-class on a smooth projective variety is algebraic.
\end{theorem}
